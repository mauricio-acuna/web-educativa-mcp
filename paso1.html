import sharp from 'sharp';
import path from 'path';

async function optimizeImages() {
  const images = await glob('src/assets/images/original/**/*.{jpg,png}');
  
  for (const image of images) {
    const outputWebP = image
      .replace('original', 'optimized')
      .replace(/\.(jpg|png)$/, '.webp');
    
    await sharp(image)
      .webp({ quality: 80 })
      .resize(1920, null, {
        withoutEnlargement: true,
        fit: 'inside'
      })
      .toFile(outputWebP);
      
    // Crear fallback JPEG optimizado
    const outputJPG = outputWebP.replace('.webp', '.jpg');
    await sharp(image)
      .jpeg({ quality: 85, progressive: true })
      .resize(1920, null, {
        withoutEnlargement: true,
        fit: 'inside'
      })
      .toFile(outputJPG);
  }
}
